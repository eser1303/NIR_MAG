\begin{frame}{Постановка задачи}
\begin{block}{Проблема}
	Для определения потенциально возможных финансовых рисков необходим мониторинг проектов в сфере жилищного строительства.
\end{block}
\begin{block}{Оценка даты фактической готовности}
	Предлагается строить прогноз ключевого макропараметра объекта жилищного строительства~--- даты фактической готовности.
\end{block}
\begin{block}{Предложение}
	Проанализировать признаковое описание объектов строительства и построить эффективную регрессионную модель, способную предсказывать дату фактической готовности.
\end{block}


\end{frame}

\begin{frame}{Ядро модели}
	\begin{block}{Линейная регрессия}
		$$
		\frac{1}{l}\sum\limits_{i=1}^{l}(\langle\omega, x_i\rangle - y_i)^2 \rightarrow \min_{\omega}.
		$$
	\end{block}
	\begin{block}{Градиентный бустинг}
	$$
	a_N(x) = \sum\limits_{n=0}^{N}\gamma_nb_n(x)-\text{взвешенная сумма базовых алгоритмов}
	$$
	$$
	\sum\limits_{i=1}^{l}L(y_i, a_{N-1}(x_i)+\gamma_Nb_N(x_i)) \rightarrow \min_{b_N, \gamma_N}.
	$$
\end{block}	
\end{frame}
%-----------------------------------------------------------------------------------------------------

\begin{frame}{Метрики оценки качества}
	
	\begin{block}{MSE}
		$$
			MSE(a, X) = \frac{1}{l}\sum\limits_{i=1}^{l}{(a(x_i)-y_i)}^2
		$$
	\end{block}
	
	\begin{block}{$R^2$}
		$$
		R^2(a, X) = 1 - \frac{\sum_{i=1}^{l}{(a(x_i)-y_i)}^2}{\sum_{i=1}^{l}{(y_i-\bar{y})}^2}
		$$	
	\end{block}

	\begin{block}{MAE}
		$$
		MAE(a,X) = \frac{1}{l}\sum\limits_{i=1}^{l}|a(x_i)-y_i|
		$$
	\end{block}

	\begin{block}{MAPE}
		$$
		MAPE(a, X) = 100\% \times \frac{1}{l}\sum\limits_{i=1}^{l}\frac{|a(x_i)-y_i|}{|y_i|}
		$$
	\end{block}
\end{frame}

\begin{frame}{Экономический эффект}	
	\begin{block}{Необходимость мониторинга}
		\begin{itemize}
			\item Риски только по одному объекту могут составлять более 25 млрд. руб.
			\item Ручной мониторинг отделом из 10 специалистов обойдется порядка 30 млн. руб ежегодно.
			\item Отношение составляет менее промилле от потенциальной суммы риска только по одному проекту.
		\end{itemize}
	\end{block}

\begin{block}{Плюсы автоматизации}
	\begin{itemize}
		\item Возможность сэкономить около 80\% трудового ресурса аналитиков
		\item Стоимость внедрения решения не превышает 5 млн. руб. при существовании возможности масштабирования.
		\item Стоимость обслуживания около 1 млн. ежегодно с учетом всего жизненного цикла.
	\end{itemize}
	
\end{block}
